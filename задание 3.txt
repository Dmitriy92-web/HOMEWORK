<?php
    $a = 5;
    $b = '05';
    var_dump($a == $b);         // Почему true? Потому что сравнение идет по заначению а не по типу а 5 = 05
    var_dump((int)'012345');     // Почему 12345? Потому что 0 не является целым числом
    var_dump((float)123.0 === (int)123.0); // Почему false? Сравнение идет по значению и по типу, типы разные один число с плавающей точкой, другое целое число
    var_dump((int)0 === (int)'hello, world'); // Почему true? Так как для текста значение int выдаст 0 и оно совпадет и по типу и по значению
?>
